-- 可在此添加 Lua 脚本进行自定义排版逻辑
-- 例如：批量处理参考文献、动态生成图表编号等
function us_to_ordinal(n)
  -- 英文序数词转换（备用）
  local s = tostring(n)
  if s:match("[^1]1$") then return s .. "st"
  elseif s:match("[^1]2$") then return s .. "nd"
  elseif s:match("[^1]3$") then return s .. "rd"
  else return s .. "th"
  end
end
\end{luacode}
